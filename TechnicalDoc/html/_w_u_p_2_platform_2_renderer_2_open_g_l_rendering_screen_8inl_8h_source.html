396 void OpenGLRenderingScreen::ScreenShot(char * filename)
397 {
398  int windowWidth = mySizeX;
399  int windowHeight = mySizeY;
400 
401  HGLRC hRC = wglGetCurrentContext();
402 
403  char* bmpBuffer = (char*)new char[windowWidth*windowHeight * 3];
404  if (!bmpBuffer)
405  return;
406 
407  glReadPixels((GLint)0, (GLint)0,
408  (GLint)windowWidth, (GLint)windowHeight,
409  GL_BGR_EXT, GL_UNSIGNED_BYTE, bmpBuffer);
410 
411  char* paddedBuffer = bmpBuffer;
412  int paddedLineSize = windowWidth * 3;
413  // padding line
414  if ((windowWidth * 3) & 3) // need padding
415  {
416  paddedLineSize += 4 - ((windowWidth * 3) & 3);
417  paddedBuffer = (char*)new char[paddedLineSize*windowHeight];
418 
419  // copy
420  char* imgread = bmpBuffer;
421  char* imgwrite = paddedBuffer;
422  int i;
423  for (i = 0; i < windowHeight; i++)
424  {
425  memcpy(imgwrite, imgread, windowWidth * 3);
426  imgwrite += paddedLineSize;
427  imgread += windowWidth * 3;
428  }
429  }
430  else
431  {
432  bmpBuffer = 0;
433  }
434  SmartPointer<FileHandle> filePtr = Platform_fopen(filename, "wb");
435  if (!filePtr)
436  return;
437 
438  BITMAPFILEHEADER bitmapFileHeader;
439  BITMAPINFOHEADER bitmapInfoHeader;
440 
441  bitmapFileHeader.bfType = 0x4D42; //"BM"
442  bitmapFileHeader.bfSize = paddedLineSize*windowHeight;
443  bitmapFileHeader.bfReserved1 = 0;
444  bitmapFileHeader.bfReserved2 = 0;
445  bitmapFileHeader.bfOffBits =
446  sizeof(BITMAPFILEHEADER) + sizeof(BITMAPINFOHEADER);
447 
448  bitmapInfoHeader.biSize = sizeof(BITMAPINFOHEADER);
449  bitmapInfoHeader.biWidth = windowWidth;
450  bitmapInfoHeader.biHeight = windowHeight;
451  bitmapInfoHeader.biPlanes = 1;
452  bitmapInfoHeader.biBitCount = 24;
453  bitmapInfoHeader.biCompression = BI_RGB;
454  bitmapInfoHeader.biSizeImage = 0;
455  bitmapInfoHeader.biXPelsPerMeter = 0; // ?
456  bitmapInfoHeader.biYPelsPerMeter = 0; // ?
457  bitmapInfoHeader.biClrUsed = 0;
458  bitmapInfoHeader.biClrImportant = 0;
459 
460  Platform_fwrite(&bitmapFileHeader, sizeof(BITMAPFILEHEADER), 1, filePtr);
461  Platform_fwrite(&bitmapInfoHeader, sizeof(BITMAPINFOHEADER), 1, filePtr);
462  Platform_fwrite(paddedBuffer, paddedLineSize*windowHeight, 1, filePtr);
463  Platform_fclose(filePtr);
464 
465  delete[] paddedBuffer;
466  if (bmpBuffer)
467  {
468  delete[] bmpBuffer;
469  }
470 }
