59 // -------------------
60 // COMPILER SWITCHES
61 // -------------------
62 // Several compiler switches are provided for library customization. Check the
63 // following section for a detailed description of each. The noteworthy ones are:
64 //
65 // DEBUG_DRAW_CXX11_SUPPORTED
66 // Enables the use of some C++11 features. If your compiler supports C++11
67 // or better, you should define this switch globally or before every inclusion
68 // of this file. If it is not defined, we try to guess it from the value of the
69 // '__cplusplus' built-in macro constant.
70 //
71 // DEBUG_DRAW_MAX_XYZ
72 // Sizes of internal intermediate buffers, which are statically allocated
73 // in the implementation. If you need to draw more primitives than the sizes
74 // of these buffers, you need to redefine them and recompile.
75 //
76 // DEBUG_DRAW_VERTEX_BUFFER_SIZE
77 // Size in dd::DrawVertex elements of the intermediate vertex buffer used
78 // to batch primitives before sending them to dd::RenderInterface. A bigger
79 // buffer will reduce the number of calls to dd::RenderInterface when drawing
80 // large sets of debug primitives.
81 //
82 // DEBUG_DRAW_OVERFLOWED(message)
83 // An error handler called if the DEBUG_DRAW_MAX_XYZ sizes overflow.
84 // By default it just prints a message to stderr.
85 //
86 // DEBUG_DRAW_USE_STD_MATH
87 // If defined to nonzero, use cmath/math.h. If you redefine it to zero before
88 // the DD implementation, it will force the use of local replacements
89 // for the library. This might be useful if you want to avoid the dependency.
90 // It is defined to zero by default (i.e. we use cmath by default).
91 //
92 // DEBUG_DRAW_XYZ_TYPE_DEFINED
93 // The compound types used by the DD library can also be customized.
94 // By default, ddVec3 and ddMat4x4 are plain C-arrays, but you can
95 // redefine them to use your own classes or structures (see below).
96 // ddStr is by default a std::string, but you can redefine it to
97 // a custom string type if necessary. The only requirement is that
98 // it provides a 'c_str()' method returning a null terminated const char* string.
99 //
100 // DEBUG_DRAW_STR_DEALLOC_FUNC(str)
101 // If you define a custom string type for ddStr and it requires some
102 // extra cleanup, you might define this function macro to perform the cleanup.
103 // It is called by dd::clear() and dd::shutdown() on every instance of the
104 // internal DebugString buffer.
105 //
106 // -------------------
107 // MEMORY ALLOCATION
108 // -------------------
109 // Debug Draw will only perform one memory allocation during startup to decompress
110 // the built-in glyph bitmap used for debug text rendering. All the vertex buffers
111 // and intermediate draw/batch buffers used internally are declared as static
112 // C-style arrays with fixed size.
113 //
114 // Memory allocation & deallocation for the glyph bitmap decompression will be done via:
115 //
116 // DD_MALLOC(size)
117 // DD_MFREE(ptr)
118 //
119 // These two macros can be redefined if you'd like to supply you own memory allocator.
120 // By default, they are defined to use std::malloc and std::free, respectively.
121 // Note: If you redefine one, you must also provide the other!
122 //
123 // --------------------------------
124 // INTERFACING WITH YOUR RENDERER
125 // --------------------------------
126 // Debug Draw doesn't touch on any renderer-specific aspects or APIs, instead you provide
127 // DD with all of it's rendering needs via the dd::RenderInterface abstract class.
128 //
129 // See the definition of dd::RenderInterface for the details. Not all methods are
130 // required. In fact, you could also implement a full no-op RenderInterface that
131 // disables debug drawing by simply inheriting from RenderInterface and not overriding
132 // any of the methods (or even easier, call dd::initialize(nullptr) to make everything a no-op).
133 //
134 // For examples on how to implement your own RenderInterface, see the accompanying samples.
135 // You can also find them in the source code repository for this project:
136 // https://github.com/glampert/debug-draw
137 //
138 // ------------------
139 // CONVENTIONS USED
140 // ------------------
141 // Points and lines are always specified in world-space positions. This also
142 // applies to shapes drawn from lines, like boxes, spheres, cones, etc.
143 //
144 // 2D screen-text is in screen-space pixels (from 0,0 in the upper-left
145 // corner of the screen to screen_width-1 and screen_height-1).
146 // RenderInterface::drawGlyphList() also receives vertexes in screen-space.
147 //
148 // We make some usage of matrices for things like the projected text labels.
149 // Matrix layout used is column-major and vectors multiply as columns.
150 // This is the convention normally used by standard OpenGL.
151 //
152 // C++ Exceptions are not used. Little error checking is provided or
153 // done inside the library. We favor simpler, faster and easier to maintain
154 // code over more sophisticated error handling. The rationale is that a
155 // debug drawing API doesn't have to be very robust, since it won't make
156 // into the final release executable in most cases.
157 //
158 // END OF DOCUMENTATION

1528  int lzwDecompress(const void * compressedData, int compressedSizeBytes,
1529  int compressedSizeBits, void * uncompressedData,
1530  int uncompressedSizeBytes)
1531  {
1532  if (compressedData == DD_NULL || uncompressedData == DD_NULL)
1533  {
1534  return 0;
1535  }
1536  if (compressedSizeBytes <= 0 || compressedSizeBits <= 0 || uncompressedSizeBytes <= 0)
1537  {
1538  return 0;
1539  }
1540 
1541  int code = LzwNil;
1542  int prevCode = LzwNil;
1543  int codeBitsWidth = LzwStartBits;
1544  int firstByte = 0;
1545  int bytesDecoded = 0;
1546 
1547  const UByte * compressedPtr = reinterpret_cast<const UByte *>(compressedData);
1548  UByte * uncompressedPtr = reinterpret_cast< UByte *>(uncompressedData);
1549 
1550  // We'll reconstruct the dictionary based on the bit stream codes.
1551  LzwBitStreamReader bitStream(compressedPtr, compressedSizeBytes, compressedSizeBits);
1552  LzwDictionary dictionary;
1553 
1554  // We check to avoid an overflow of the user buffer.
1555  // If the buffer is smaller than the decompressed size, we
1556  // break the loop and return the current decompression count.
1557  while (bitStream.numBitsRead < bitStream.sizeInBits)
1558  {
1559  if (codeBitsWidth > LzwMaxDictBits)
1560  {
1561  break;
1562  }
1563  code = bitStream.readBits(codeBitsWidth);
1564 
1565  if (prevCode == LzwNil)
1566  {
1567  if (!lzwOutputByte(code, uncompressedPtr, uncompressedSizeBytes, bytesDecoded))
1568  {
1569  break;
1570  }
1571  firstByte = code;
1572  prevCode = code;
1573  continue;
1574  }
1575 
1576  if (code >= dictionary.size)
1577  {
1578  if (!lzwOutputSequence(dictionary, prevCode, uncompressedPtr,
1579  uncompressedSizeBytes, bytesDecoded, firstByte))
1580  {
1581  break;
1582  }
1583  if (!lzwOutputByte(firstByte, uncompressedPtr, uncompressedSizeBytes, bytesDecoded))
1584  {
1585  break;
1586  }
1587  }
1588  else
1589  {
1590  if (!lzwOutputSequence(dictionary, code, uncompressedPtr,
1591  uncompressedSizeBytes, bytesDecoded, firstByte))
1592  {
1593  break;
1594  }
1595  }
1596 
1597  if (!dictionary.add(prevCode, firstByte))
1598  {
1599  break;
1600  }
1601 
1602  if (dictionary.flush(codeBitsWidth))
1603  {
1604  prevCode = LzwNil;
1605  }
1606  else
1607  {
1608  prevCode = code;
1609  }
1610  }
1611 
1612  return bytesDecoded;
1613  }
