541 template<typename... T>
542 inline bool CoreModifiable::EmitSignal(const KigsID& SignalID, T&&... params)
543 {
544  if (!isFlagAllowChanges()) return false;
545  std::unique_lock<std::recursive_mutex> lk{ GetMutex() };
546  if (!isFlagAllowChanges()) return false;
547 
548  if (!mLazyContent) return false;
549 
550  auto it = GetLazyContent()->mConnectedTo.find(SignalID);
551  if(it != GetLazyContent()->mConnectedTo.end())
552  {
553  std::vector<std::pair<KigsID, CoreModifiable*>> copy = it->second;
554  lk.unlock();
555 
556  PackCoreModifiableAttributes attr(nullptr);
557  int expander[]
558  {
559  (attr << std::forward<T>(params), 0)...
560  };
561  (void)expander;
562  auto& attr_list = (kstl::vector<CoreModifiableAttribute*>&)attr;
563 
564  for(auto p : copy)
565  {
566  if (!p.second->isFlagAllowChanges()) continue;
567  std::unique_lock<std::recursive_mutex> lk{ p.second->GetMutex() };
568  if (!p.second->isFlagAllowChanges()) continue;
569  p.second->CallMethod(p.first, attr_list, nullptr, this);
570  }
571  return true;
572  }
573 
574  return false;
575 }
