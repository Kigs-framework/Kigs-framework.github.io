40  inline bool IntersectionFastRayBBoxTest(const Point3D &RayOrigin,
41  const Vector3D &RayDirection, kfloat MaxDist,
42  const Point3D &BBoxMin, const Point3D &BBoxMax)
43  {
45  Vector3D center(BBoxMin);
46  center += BBoxMax;
47  center *= 0.5f;
48 
50  Vector3D diag(BBoxMax);
51  diag -= BBoxMin;
52  diag *= 0.5f;
53 
54  kfloat r = NormSquare(diag);
55 
56  Vector3D Q(center);
57  Q -= RayOrigin;

60 
61  if ((v < 0.0f) && (c > r))
62  {
63  return false;

65  else if ((v > 0.0f) && ((c - MaxDist*MaxDist) > r))
66  {
67  return false;

79  inline bool IntersectionRaySphere(const Point3D &RayOrigin,
80  const Vector3D &RayDirection,
81  const Point3D &SphereCenter,
82  const kfloat SphereRadius,
83  kfloat &FirstIntersectionDistance)

344  auto ad = v3f(fabs(d.x), fabs(d.y), fabs(d.z));
345 
346  if (fabsf(c[0]) > e[0] + ad[0])

349  return false;

351  return false;
352 
353  if (fabsf(d[1] * c[2] - d[2] * c[1]) > e[1] * ad[2] + e[2] * ad[1] + FLT_EPSILON)
354  return false;
