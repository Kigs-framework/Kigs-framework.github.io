362 nodeInfo OctreeBase<BaseType>::getVoxelAt(const v3i& coordinate, unsigned int maxDepth)
363 {
364  nodeInfo result;
365  result.coord = coordinate;
366  result.level = 0;
367  result.node = mRootNode;
368  int currentDecal = mMaxDepth;
369  do
370  {
371  if (result.node->isLeaf())
372  {
373  break;
374  }
375 #ifdef _DEBUG
376  if (result.level >= mMaxDepth)
377  {
378  result.node = nullptr;
379  printf("should not occur\n");
380  break;
381  }
382 #endif
383  if (result.level >= maxDepth)
384  {
385  break;
386  }
387 
388  unsigned int index = ((coordinate.x >> currentDecal) & 1) | (((coordinate.y >> currentDecal) & 1) << 1) | (((coordinate.z >> currentDecal) & 1) << 2);
389  result.node = result.node->getChild(index);
390  result.level++;
391  currentDecal--;
392 
393  } while (1);
394 
395  setValidCubeCenter(result.coord, currentDecal);
396  return result;

457 {
458 
459  int dpos = 1 << (mMaxDepth - (node.level + 1));
460 
461  // check all sons
462  for (int c = 0; c < 8; c++)
463  {
464  if ((c & mMaskTest) == mMaskResult)
465  {
466  nodeInfo toAdd;
467  toAdd.node = node.node->getChild(c);
468  toAdd.level = node.level + 1;
469 
470  // add block so childCoord is not pushed for recursion
471  {
472  v3i childCoord = node.coord;
473  childCoord.x += (((c & 1) << 1) - 1) * dpos;
474  childCoord.y += ((c & 2) - 1) * dpos;
475  childCoord.z += (((c & 4) >> 1) - 1) * dpos;
476 
477  toAdd.coord = childCoord;
478  }
479 
480  if (toAdd.node->isLeaf())

493 template<typename BaseType>
494 void OctreeBase<BaseType>::recursiveFloodFill::run(const nodeInfo& startPos)
495 {
496  // set current node as "treated"
497  startPos.node->setBrowsingFlag(mCurrentBrowsingFlag);
498 
499  // for each adjacent node
500  for (int dir = 0; dir < 6; dir++)
501  {
502  // get adjacent node
503  nodeInfo n = getVoxelNeighbour(startPos, dir);
504 
505  if (n.node == nullptr) // TODO => outside of this octree -> should check other octrees
506  {
507  continue;
508  }
509  if (n.node->getBrowsingFlag() == mCurrentBrowsingFlag) // already treated continue
510  {
511  continue;
512  }
513 
514  std::vector< nodeInfo> child;
515  if (n.node->isLeaf()) // if this node is a leaf, then this is the only one to treat
516  {
517  child.push_back(n);
518  }
519  else // else get all sons on the correct side of n
520  {
521  recurseVoxelSideChildren r(mInvDir[dir], *this, &child);
522  r.run(n);
523  }
524 
525  // for all the found nodes, check if they need to be added to visible list or to be flood fill
526  for (auto& c : child)
527  {
528  if (c.node->getBrowsingFlag() != mCurrentBrowsingFlag)
529  {
530  if (c.node->isEmpty()) // node is empty, recurse flood fill
531  {
532  run(c);
533  continue;
534  }
535  else // add this node to visibility list
536  {
537  c.node->setBrowsingFlag(mCurrentBrowsingFlag);
538  (*mNotEmptyList).push_back(c);
539  }
540  }
541  }
542  }
