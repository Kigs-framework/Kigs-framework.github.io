191  // This holds the range we intend to operate on, or which we intend
192  // to become the new selection. Start with the current selection.
193  CoreTextRange range = selection;
194 
195  // For the purpose of this sample, we will support only the left and right
196  // arrow keys and the backspace key. A more complete text edit control
197  // would also handle keys like Home, End, and Delete, as well as
198  // hotkeys like Ctrl+V to paste.
199  //
200  // Note that this sample does not properly handle surrogate pairs
201  // nor does it handle grapheme clusters.
202 
203  switch (args.VirtualKey())
204  {
205  // Backspace
206  case winrt::Windows::System::VirtualKey::Back:
207  // If there is a selection, then delete the selection.
208  if (HasSelection())
209  {
210  // Set the text in the selection to nothing.
211  ReplaceText(range, L"");
212  }
213  else
214  {
215  // Delete the character to the left of the caret, if one exists,
216  // by creating a range that encloses the character to the left
217  // of the caret, and setting the contents of that range to nothing.
218  range.StartCaretPosition = std::max(0, range.StartCaretPosition - 1);
219  ReplaceText(range, L"");
220  }
221  break;
222  // Enter
223  case winrt::Windows::System::VirtualKey::Enter:
224  ReplaceText(range, L"\n");
225  break;
226  // Left arrow
227  case winrt::Windows::System::VirtualKey::Left:
228  // If the shift key is down, then adjust the size of the selection.
229  if (window.GetKeyState(winrt::Windows::System::VirtualKey::Shift) == winrt::Windows::UI::Core::CoreVirtualKeyStates::Down)
230  {
231  // If this is the start of a selection, then remember which edge we are adjusting.
232  if (!HasSelection())
233  {
234  extendingLeft = true;
235  }
236 
237  // Adjust the selection and notify CoreTextEditContext.
238  AdjustSelectionEndpoint(-1);
239  }
240  else
241  {
242  // The shift key is not down. If there was a selection, then snap the
243  // caret at the left edge of the selection.
244  if (HasSelection())
245  {
246  range.EndCaretPosition = range.StartCaretPosition;
247  selection_need_set = true;
248  SetSelectionAndNotify(range);
249  }
250  else
251  {
252  // There was no selection. Move the caret left one code unit if possible.
253  range.StartCaretPosition = std::max(0, range.StartCaretPosition - 1);
254  range.EndCaretPosition = range.StartCaretPosition;
255  selection_need_set = true;

257  }

384  void ReplaceText(CoreTextRange modifiedRange, std::wstring newtext)
385  {
386  // Modify the internal text store.
387  auto old_text = text;
388  text = text.substr(0, modifiedRange.StartCaretPosition) +
389  newtext +
390  text.substr(modifiedRange.EndCaretPosition);
391 
392  if (text == old_text) return;
393  utf8_text = to_utf8(text);
394 
395  // Move the caret to the end of the replacement text.
396  selection.StartCaretPosition = modifiedRange.StartCaretPosition + newtext.size();
397  selection.EndCaretPosition = selection.StartCaretPosition;
398 
399  // Update the selection of the edit context. There is no need to notify the system
400  // of the selection change because we are going to call NotifyTextChanged soon.
401  SetSelection(selection);
402  text_need_set = true;
403  // Let the CoreTextEditContext know what changed.
404  editContext.NotifyTextChanged(modifiedRange, text.size(), selection);
405  }
