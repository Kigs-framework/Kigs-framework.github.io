21 
22 Compiling
23 ---------
24 
25 * Windows (MSVC) - add lib/Remotery.c and lib/Remotery.h to your program. Set include
26  directories to add Remotery/lib path. The required library ws2_32.lib should be picked
27  up through the use of the #pragma comment(lib, "ws2_32.lib") directive in Remotery.c.
28 
29 * Mac OS X (XCode) - simply add lib/Remotery.c and lib/Remotery.h to your program.
30 
31 * Linux (GCC) - add the source in lib folder. Compilation of the code requires -pthreads for
32  library linkage. For example to compile the same run: cc lib/Remotery.c sample/sample.c
33  -I lib -pthread -lm
34 
35 You can define some extra macros to modify what features are compiled into Remotery. These are
36 documented just below this comment.
37

86 // How many times we spin data back and forth between CPU & GPU
87 // to calculate average RTT (Roundtrip Time). Cannot be 0.
88 // Affects OpenGL & D3D11
89 #ifndef RMT_GPU_CPU_SYNC_NUM_ITERATIONS
90 #define RMT_GPU_CPU_SYNC_NUM_ITERATIONS 16
91 #endif
92 
93 // Time in seconds between each resync to compensate for drifting between GPU & CPU timers,
94 // effects of power saving, etc. Resyncs can cause stutter, lag spikes, stalls.
95 // Set to 0 for never.
96 // Affects OpenGL & D3D11
97 #ifndef RMT_GPU_CPU_SYNC_SECONDS
98 #define RMT_GPU_CPU_SYNC_SECONDS 30
99 #endif
100 
101 // Whether we should automatically resync if we detect a timer disjoint (e.g.
102 // changed from AC power to battery, GPU is overheating, or throttling up/down
103 // due to laptop savings events). Set it to 0 to avoid resync in such events.
104 // Useful if for some odd reason a driver reports a lot of disjoints.
105 // Affects D3D11
106 #ifndef RMT_D3D11_RESYNC_ON_DISJOINT
107 #define RMT_D3D11_RESYNC_ON_DISJOINT 1

354 // Struture to fill in to modify Remotery default settings
355 typedef struct rmtSettings
356 {
357  // Which port to listen for incoming connections on
358  rmtU16 port;
359 
360  // When this server exits it can leave the port open in TIME_WAIT state for
361  // a while. This forces subsequent server bind attempts to fail when
362  // restarting. If you find restarts fail repeatedly with bind attempts, set
363  // this to true to forcibly reuse the open port.
364  rmtBool reuse_open_port;
365 
366  // Only allow connections on localhost?
367  // For dev builds you may want to access your game from other devices but if
368  // you distribute a game to your players with Remotery active, probably best
369  // to limit connections to localhost.
370  rmtBool limit_connections_to_localhost;
371 
372  // How long to sleep between server updates, hopefully trying to give
373  // a little CPU back to other threads.
374  rmtU32 msSleepBetweenServerUpdates;
375 
376  // Size of the internal message queues Remotery uses
377  // Will be rounded to page granularity of 64k
378  rmtU32 messageQueueSizeInBytes;
379 
380  // If the user continuously pushes to the message queue, the server network
381  // code won't get a chance to update unless there's an upper-limit on how
382  // many messages can be consumed per loop.
383  rmtU32 maxNbMessagesPerUpdate;
384 
385  // Callback pointers for memory allocation
386  rmtMallocPtr malloc;
387  rmtReallocPtr realloc;
388  rmtFreePtr free;
389  void* mm_context;
390 
391  // Callback pointer for receiving input from the Remotery console
392  rmtInputHandlerPtr input_handler;
393 
394  // Context pointer that gets sent to Remotery console callback function
395  void* input_handler_context;
396 
397  rmtPStr logFilename;
398 } rmtSettings;
